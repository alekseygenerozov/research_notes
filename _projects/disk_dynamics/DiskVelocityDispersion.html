import h5py
from scipy.interpolate import interp1d
%pylab inline

fig,ax=plt.subplots(figsize=(10,9))
ax.set_xlim(1., 500.)
ax.set_ylim(0.01, 0.11)
ax.set_xlabel('Time [Orbits]')
names=np.genfromtxt('sim_data/circ_disk_200/names', dtype=str)

t_std=np.arange(0., 499.*2.*np.pi, 0.2*np.pi)
t_std[0]=2.0e-15
sig_std=np.empty([len(names), len(t_std)])
sig_std_e=np.empty([len(names), len(t_std)])


for ii,name in enumerate(names):
    name1=name.replace('.bin','_sigs.h5')
    name2=name.replace('.bin','_times')
    f=h5py.File('sim_data/circ_disk_200/'+name1)
    times=np.genfromtxt('sim_data/circ_disk_200/'+name2)
    sigs=(f['bin_0'][...]['col2'])
    sigs_e=(f['bin_0'][...]['col3'])
    sig_std[ii]=interp1d(times, sigs)(t_std)
    sig_std_e[ii]=interp1d(times, sigs_e)(t_std)

loglog(t_std/(2.*np.pi), np.mean(sig_std, axis=0), label=r'$\sigma_z$')
loglog(t_std/(2.*np.pi), 0.5*np.mean(sig_std_e, axis=0), label=r'$\sigma_{xy}/2$')

ax.legend(title='1<=a<1.2')

def evolve (m , model_time, z=0.02, t_inc=2.) :
    '''
    Evolve stars with masses given by m, for time model_time with the 
    parameterized stellar evolution code SSE. 
    
    Keyword argument z specifies the stars' metallicity. 
    
    Particle properties are writen to hdf5 file <stellar.hdf5> at t=2*t_inc^n Myr, where n is an integer
    '''
    stellar = SSE ()
    stellar . parameters . metallicity = z
    part = Particles ( mass = m )
    stellar . particles . add_particles ( part )
    stellar . commit_particles ()
    
    initial_luminosity = stellar . particles . luminosity
    time = 0 | units . Myr

    idx=0
    m_initial=stellar.particles.mass
    time = 2.0|units.Myr
    times = []
    while stellar . particles [0]. age < model_time :
        print time
        time = time * 2.**0.5
        stellar.evolve_model ( time )
        times.append(time.value_in(units.s))
        #Extracting physical properties of star.
        io.write_set_to_file(
            stellar.particles,
           'stellar.hdf5',
           'hdf5')
 
        idx=idx+1
    stellar.stop()

    return np.array(times)

def rndm(a, b, g, size=1):
    """Generate size random number from pdf(x)\propto x^(g-1)  for a<=x<=b"""
    r = np.random.random(size=size)	

    ag, bg = a**g, b**g
    return (ag + (bg - ag)*r)**(1./g)
